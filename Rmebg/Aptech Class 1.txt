from rembg import remove
from PIL import Image
input_path_1 = 'image1.jpg'
output_path_1 = 'image1_removed.png'
input_image_1 = Image.open(input_path_1)
output_image_1 = remove(input_image_1)
output_image_1.save(output_path_1)
print("Background removed for image1.jpg")
